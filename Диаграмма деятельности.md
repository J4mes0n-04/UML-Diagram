# Пример диаграммы деятельности. Mermaid

## Процесс обработки заказа в интернет-магазине (приме)

```mermaid

    flowchart TD

    Start([Начало]) --> A[Пользователь оформляет заказ]
    A --> B{Проверка наличия товара}
    
    B -- Есть в наличии --> C[Зарезервировать товар]
    B -- Нет в наличии --> D[Уведомить клиента]
    D --> E[Предложить альтернативу]
    E --> F{Клиент согласен?}
    
    F -- Да --> C
    F -- Нет --> G[Отменить заказ]
    G --> End1([Конец процесса])
    
    C --> H[Сформировать счет]
    H --> I[Отправить на оплату]
    I --> J{Оплата получена?}
    
    J -- Да --> K[Подготовить к отправке]
    J -- Нет --> L[Напомнить об оплате]
    L --> M{Оплата в течение 3 дней?}
    
    M -- Да --> K
    M -- Нет --> N[Аннулировать заказ]
    N --> End2([Конец процесса])
    
    K --> O[Передать в службу доставки]
    O --> P[Уведомить клиента]
    P --> Q[Доставить заказ]
    Q --> R{Заказ получен?}
    
    R -- Да --> S[Завершить заказ]
    R -- Нет --> T[Выяснить причину]
    T --> U[Решить проблему]
    U --> Q
    
    S --> End3([Конец процесса])
    
    style Start fill:#9f9
    style End1 fill:#f99
    style End2 fill:#f99
    style End3 fill:#9f9
